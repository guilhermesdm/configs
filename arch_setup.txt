[Defining partitions]

> parted /dev/sda
> mkpart "EFI system partition" fat32 1MiB 1025M
> set 1 esp on
> mkpart "swap partition" linux-swap 1025M 32G # same as ram
> mkpart "root partition" ext4 32G 100%
> quit

Checking with ```lsblk /dev/sda```

[Format partitions]

Boot: ```mkfs.fat -F32 /dev/sda1```
Swap: ```mkswap /dev/sda2 && swapon /dev/sda2```
Root: ```mkfs -t ext4 /dev/sda3```

[Mount partitions]

```mount /mnt /dev/sda3```
```mkdir /mnt/boot && mount /dev/sda1 /mnt/boot```

[Install base packages]

pacstrap /mnt base base-devel linux linux-firmware neovim zsh git 
genfstab -U /mnt >> /mnt/etc/fstab

[Setting locale]

Enter chroot
arch-chroot /mnt
ln -sf /usr/share/zoneinfo/America/Sao_Paulo /etc/localtime
hwclock --systohc
printf "\nen_US.UTF-8 UTF-8\npt_BR.UTF-8 UTF-8\n" >> /etc/locale.gen
locale-gen
echo LANG=en_US.UTF-8 > /etc/locale.conf
systemctl enable systemd-timesyncd.service

[Network]

echo guilherme > /etc/hostname
vim /etc/hosts
    # 127.0.0.1 localhost

    # 127.0.1.1 guilherme

pacman -S dhcpcd dialog bluez bluez-utils networkmanager network-manager-applet resolvconf
# wireless_tools - optional
